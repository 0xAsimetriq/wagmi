# usePrepareContractTransaction

Hook for preparing a contract transaction to be sent via [`useContractWritePrepared`](/docs/hooks/useContractWritePrepared).

Eagerly fetches the parameters required for sending a contract transaction such as the gas limit.

```tsx
import { usePrepareContractTransaction } from 'wagmi'
```

## Usage

`usePrepareContractTransaction` gives back a "prepared request" to be sent through to `useContractWritePrepared`.

```tsx {4-8}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest, error } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
  })
  const { write } = useContractWritePrepared({
    preparedRequest,
  })

  return (
    <>
      <button disabled={!write} onClick={() => write?.()}>
        Feed
      </button>
      {error && (
        <div>An error occurred preparing the transaction: {error.message}</div>
      )}
    </>
  )
}
```

Note: The `write` function will be undefined if the request has not been prepared.

## Return value

```tsx
{
  data?: PopulatedTransaction
  error?: Error
  isIdle: boolean
  isLoading: boolean
  isFetching: boolean
  isSuccess: boolean
  isError: boolean
  isFetched: boolean
  isRefetching: boolean
  refetch: (options: {
    throwOnError: boolean
    cancelRefetch: boolean
  }) => Promise<PopulatedTransaction>
  status: 'idle' | 'error' | 'loading' | 'success'
}
```

## Configuration

### addressOrName

Contract address or ENS name.

```tsx {5}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
  })
}
```

### contractInterface

Contract ABI in JSON or JS object format. An [ethers Interface](https://docs.ethers.io/v5/api/utils/abi/interface) is also allowed.

```tsx {6}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
  })
}
```

### functionName

Name of function to call.

```tsx {7}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
  })
}
```

### args (optional)

Arguments to pass to function call. Accepts `any | any[]`.

```tsx {8}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
    args: [],
  })
}
```

### overrides (optional)

[Overrides](https://docs.ethers.io/v5/api/contract/contract/#Contract-functionsCall) to pass to function call.

```tsx {8-11}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
    overrides: {
      from: '0xA0Cf798816D4b9b9866b5330EEa46a18382f251e',
      value: ethers.utils.parseEther('0.01'),
    },
  })
}
```

### cacheTime (optional)

Time (in ms) which the prepared request should remain in the cache.

```tsx {8}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
    cacheTime: 2_000,
  })
}
```

### enabled (optional)

Set this to `false` to disable this query from automatically running. Defaults to `true`.

```tsx {8}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
    enabled: false,
  })
}
```

### staleTime (optional)

Time (in ms) after prepared request is considered stale. If set to `Infinity` the data will never be considered stale.

```tsx {8}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
    staleTime: 2_000,
  })
}
```

### suspense (optional)

Set this to `true` to enable suspense mode.

```tsx {8}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
    suspense: true,
  })
}
```

### onSuccess (optional)

Function to invoke when fetching is successful.

```tsx {8-10}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
    onSuccess(data) {
      console.log('Success', data)
    },
  })
}
```

### onError (optional)

Function to invoke when an error is thrown while fetching new data.

```tsx {8-10}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
    onError(error) {
      console.log('Error', error)
    },
  })
}
```

### onSettled (optional)

Function to invoke when fetching is settled (either successfully fetched, or an error has thrown).

```tsx {8-10}
import { usePrepareContractTransaction } from 'wagmi'

function App() {
  const { data: preparedRequest } = usePrepareContractTransaction({
    addressOrName: '0xecb504d39723b0be0e3a9aa33d646642d1051ee1',
    contractInterface: wagmigotchiABI,
    functionName: 'feed',
    onSettled(data, error) {
      console.log('Settled', { data, error })
    },
  })
}
```
